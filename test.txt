for _, v in pairs(game:GetService("ServerScriptService"):GetChildren()) do
    if v:IsA("Script") or v:IsA("ModuleScript") then
        local scriptSource = decompile(v) -- Decompiles the script
        if scriptSource then
            writefile("SavedScripts/" .. v.Name .. ".lua", scriptSource) -- Saves the script locally
        end
    end
end
